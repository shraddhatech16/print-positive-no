# print-positive-no
list1 = [12, -7, 5, 64, -14]
num = 0
  
# using while loop     
while(num < len(list1)):
      
    # checking condition
    if list1[num] >= 0:
        print(list1[num], end = " ")
      
    # increment num 
    num += 1![positive](https://user-images.githubusercontent.com/83664806/131239049-ba71dffd-c1ad-4511-a06a-f193b64036bf.PNG)
